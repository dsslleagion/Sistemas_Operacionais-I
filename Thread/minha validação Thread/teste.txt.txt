        MinhaThread thread2 = new MinhaThread("Neymar", 1004);
        MinhaThread thread3 = new MinhaThread("Messi", 1003);

        System.out.println("Cristiano Ganhou!");
        System.out.println("Messi Chegou em 2º");

    }
}
